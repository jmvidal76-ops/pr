CREATE TABLE VISTAS
( 
  ID_VISTA INT IDENTITY(1,1),
  NOMBRE VARCHAR(100) NOT NULL,
  CODIGO VARCHAR(50) NOT NULL,
  CONSTRAINT PK_VISTAS PRIMARY KEY (ID_VISTA)
)

CREATE TABLE PERMISOS
( 
  ID_VISTA INT NOT NULL,
  ID_ROL INT NOT NULL,
  CONSULTA INT NOT NULL DEFAULT 1,
  EDICION INT NOT NULL DEFAULT 1,
  CONSTRAINT PK_PERMISOS_PK PRIMARY KEY (ID_VISTA,ID_ROL),
  CONSTRAINT FK_PERMISOS_ROLES FOREIGN KEY (ID_ROL) REFERENCES ROLES(ID_ROL),
  CONSTRAINT FK_PERMISOS_VISTAS FOREIGN KEY (ID_VISTA) REFERENCES VISTAS(ID_VISTA)
)

CREATE TABLE MENUS
( ID_MENU INT NOT NULL,
  NOMBRE VARCHAR(100) NOT NULL,
  NIVEL INT NOT NULL DEFAULT 0,
  ORDEN INT NOT NULL DEFAULT 0,
  APLICACION VARCHAR(1) NOT NULL DEFAULT 'P',
  ID_MENU_PADRE INT NULL,
  ID_VISTA INT NOT NULL,
  CONSTRAINT PK_MENU PRIMARY KEY (ID_MENU),
  CONSTRAINT FK_MENU_MENUPADRE FOREIGN KEY (ID_MENU_PADRE) REFERENCES MENUS(ID_MENU),
  CONSTRAINT FK_MENU_VISTAS FOREIGN KEY (ID_VISTA) REFERENCES VISTAS(ID_VISTA)
)

CREATE TABLE INCIDENCIAS
( 
  ID_INCIDENCIA INT IDENTITY(1,1),
  USUARIO VARCHAR(50) NOT NULL,
  PANTALLA VARCHAR(100) NOT NULL,
  DESCRIPCION VARCHAR(2000) NOT NULL,
  APLICACION VARCHAR(1) NOT NULL DEFAULT 'P',
  CONSTRAINT PK_INCIDENCIAS PRIMARY KEY (ID_INCIDENCIA)
)


CREATE TABLE [dbo].[EQUIPO_CONSTRUCTIVO](
	[ID_EQUIPO_CONSTRUCTIVO] [int] IDENTITY(1,1) NOT NULL,
	[ID_MAQUINA] [nvarchar](100) NULL,
	[NOMBRE] [nvarchar](200) NULL,
	[CODIGO] [varchar](50) NULL,
 CONSTRAINT [PK_EQUIPO_CONSTRUCTIVO] PRIMARY KEY CLUSTERED 
(
	[ID_EQUIPO_CONSTRUCTIVO] ASC
)WITH (PAD_INDEX = OFF, STATISTICS_NORECOMPUTE = OFF, IGNORE_DUP_KEY = OFF, ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON) ON [PRIMARY]
) ON [PRIMARY]


CREATE TABLE [dbo].[POSIBLES_AVERIAS](
	[ID_POSIBLE_AVERIA] [int] IDENTITY(1,1) NOT NULL,
	[ID_EQUIPO_CONSTRUCTIVO] [int] NOT NULL,
	[DESCRIPCION] [nvarchar](300) NULL,
 CONSTRAINT [PK_POSIBLES_AVERIAS] PRIMARY KEY CLUSTERED 
(
	[ID_POSIBLE_AVERIA] ASC
)WITH (PAD_INDEX = OFF, STATISTICS_NORECOMPUTE = OFF, IGNORE_DUP_KEY = OFF, ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON) ON [PRIMARY]
) ON [PRIMARY]

GO
